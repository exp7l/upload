# goal: derive MKR-USD spot after each swap ie. the execution price of the swap, from the swap events

# from: UNIV3, MKR-ETH, 0.3% pool: https://info.uniswap.org/#/pools/0xe8c6c9227491c0a8156a0106a0204d881bb7e531
# note: this column is the price observation after each swap, not a quote given before swap
# derivation: sqrtPriceX96 = sqrt(price) * 2 ** 96, https://docs.uniswap.org/sdk/v3/guides/fetching-prices
# derivation: https://github.com/Uniswap/v3-core/blob/main/contracts/UniswapV3Pool.sol#L666
df_uni['umkr-eth'] = df_uni.apply(lambda r: r['sqrtPriceX96'] ** 2 / (2 ** (96 * 2)), axis=1)

# ETH-USD effectively is from USDC-ETH 0.3% pool, $100M+ TVL: https://github.com/Uniswap/v3-subgraph/blob/bf03f940f17c3d32ee58bd37386f26713cff21e2/src/utils/pricing.ts#L60
df_uni['ueth-usd'] = df_uni.apply(lambda r: abs(r['amountUSD'] / r['amount1']), axis=1)

# finally, we get the spot mkr-usd after the swap
df_uni['umkr-usd'] = df_uni.apply(lambda r: r['umkr-eth'] * r['ueth-usd'], axis=1)

TWAP calculation (based on accumulated prices)¶

https://uniswapv3book.com/docs/milestone_5/price-oracle/#how-uniswap-price-oracle-works

def twap(df_uni, minutes):
    def f(row):
        # (a2 - a1) / 3600
        # row.name is the timestamp, ie. the index, of the row
        a1 = df_uni.loc[row.name - timedelta(minutes=minutes)]['umkr-usd-accum']
        a2 = df_uni.loc[row.name]['umkr-usd-accum']
        return (a2 - a1) / (minutes * 60)
    return f

# filter down to the timestamps where there are swaps, for efficiency sake (there are 9m rows in total when not filtered!)
df_uni_swap_observed = df_uni[df_uni['umkr-usd'].notna()]

# **at last** calculate the 1 hour TWAP
idx = df_uni_swap_observed.index
df_filterd = df_uni_swap_observed[idx > idx.min() + timedelta(minutes=60)]

df_uni['umkr-usd-1htwap'] = df_filterd.apply(twap(df_uni, 60), axis=1)
df_uni['umkr-usd-1htwap'] = df_uni['umkr-usd-1htwap'].interpolate(method='time')

df_uni['umkr-usd-1mtwap'] = df_filterd.apply(twap(df_uni, 1), axis=1)
df_uni['umkr-usd-1mtwap'] = df_uni['umkr-usd-1mtwap'].interpolate(method='time')

# rows before 1st swap observation will be NaN, let's filter them out
df_uni = df_uni[df_uni['umkr-usd-1htwap'].notna()]

lot = 30000

# mean of 50 trials, each time simulate a $3m open market program, $30,000 each lot
# find the total over payment comparing 1h-TWAP to spot price
def print_stats(df):
    kernel = scipy.stats.gaussian_kde(df)
    trials = np.asarray([
        np.sum(kernel.resample(100)[0]) * lot
        for i in range(50)]
    )
    print(pd.Series(trials).describe())
    
print('---1 hour TWAP---')
